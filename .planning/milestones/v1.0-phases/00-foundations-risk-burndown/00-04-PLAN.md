package dev.tessera.core.bench;

import org.openjdk.jmh.results.format.ResultFormatType;
import org.openjdk.jmh.runner.Runner;
import org.openjdk.jmh.runner.options.OptionsBuilder;

import java.nio.file.*;
import java.time.format.DateTimeFormatter;
import java.time.Instant;
import java.time.ZoneOffset;

public final class JmhRunner {
    public static void main(String[] args) throws Exception {
        String dataset = System.getProperty("jmh.dataset", "100000");
        Path repoRoot = findRepoRoot();
        Path outDir = repoRoot.resolve(".planning/benchmarks");
        Files.createDirectories(outDir);
        String stamp = DateTimeFormatter.ISO_INSTANT
                .withZone(ZoneOffset.UTC)
                .format(Instant.now())
                .replace(":", "-");
        Path outFile = outDir.resolve(stamp + "-" + dataset + ".json");

        new Runner(new OptionsBuilder()
                .include("dev\\.tessera\\.core\\.bench\\..*Bench")
                .resultFormat(ResultFormatType.JSON)
                .result(outFile.toString())
                .jvmArgsAppend("-Djmh.dataset=" + dataset)
                .shouldFailOnError(true)
                .build()).run();

        // also write a stable "latest" pointer file for regression check
        Files.writeString(outDir.resolve("latest-" + dataset + ".json"), Files.readString(outFile));
        System.out.println("JMH results: " + outFile);
    }

    private static Path findRepoRoot() {
        Path p = Paths.get("").toAbsolutePath();
        while (p != null && !Files.exists(p.resolve("pom.xml").toAbsolutePath())) p = p.getParent();
        if (p == null) throw new IllegalStateException("repo root not found");
        return p;
    }

    private JmhRunner() {}
}

Create `scripts/check_regression.sh` — POSIX shell, +x:

```bash
#!/usr/bin/env bash
# Compares .planning/benchmarks/latest-<DATASET>.json against the previous
# green baseline and exits 1 if any benchmark's p95 (SampleTime) regresses
# by more than 25%.
#
# Usage: scripts/check_regression.sh <dataset>   e.g. 100000
#
# Per Phase 0 D-04. The 25% threshold is intentionally loose for the first
# baseline run — Phase 1 may tighten.

set -euo pipefail

DATASET="${1:-100000}"
THRESHOLD_PCT="${REGRESSION_THRESHOLD_PCT:-25}"
DIR=".planning/benchmarks"
LATEST="$DIR/latest-$DATASET.json"
BASELINE="$DIR/baseline-$DATASET.json"

if [[ ! -f "$LATEST" ]]; then
  echo "ERROR: no latest result at $LATEST — run JMH first" >&2
  exit 2
fi

if [[ ! -f "$BASELINE" ]]; then
  echo "INFO: no baseline at $BASELINE — seeding baseline from latest"
  cp "$LATEST" "$BASELINE"
  exit 0
fi

# Use jq to extract benchmark name -> p95 sample time
extract() {
  jq -r '.[] | select(.mode == "sample") | "\(.benchmark)\t\(.primaryMetric.scorePercentiles."95.0")"' "$1"
}

failed=0
while IFS=$'\t' read -r bench latest_p95; do
  base_p95=$(extract "$BASELINE" | awk -v b="$bench" '$1==b {print $2}')
  if [[ -z "$base_p95" ]]; then
    echo "INFO: $bench is new — skipping regression check"
    continue
  fi
  ratio=$(awk -v l="$latest_p95" -v b="$base_p95" 'BEGIN { printf "%.2f", (l - b) / b * 100 }')
  printf "%-60s baseline=%-12s latest=%-12s delta=%+s%%\n" "$bench" "$base_p95" "$latest_p95" "$ratio"
  over=$(awk -v r="$ratio" -v t="$THRESHOLD_PCT" 'BEGIN { print (r > t) ? "1" : "0" }')
  if [[ "$over" == "1" ]]; then
    echo "FAIL: $bench p95 regressed by $ratio% (threshold ${THRESHOLD_PCT}%)" >&2
    failed=1
  fi
done < <(extract "$LATEST")

exit "$failed"
```
